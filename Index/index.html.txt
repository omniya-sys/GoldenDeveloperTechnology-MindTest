<script>
let currentQuestion = 1;
let username = "";
let scores = {Analytical:0, Creative:0, Balanced:0};

function startQuiz() {
  const input = document.getElementById('username').value.trim();
  if(input === "") { alert("Please enter your name"); return; }
  username = input;
  document.getElementById('nameInputSection').style.display = 'none';
  document.getElementById('q1').style.display = 'block';
}

function selectAnswer(type) {
  scores[type]++;
  document.getElementById('q'+currentQuestion).style.display = 'none';
  currentQuestion++;
  if(currentQuestion <= 5) {
    document.getElementById('q'+currentQuestion).style.display = 'block';
  } else {
    showResult();
  }
}

function showResult() {
  let maxType = Object.keys(scores).reduce((a,b) => scores[a] > scores[b] ? a : b);

  let descriptions = {
    Analytical: `Analytical / Logical Thinker – focuses on data and careful planning.\n
Traits of suitable roles:\n
- Solve complex problems and analyze data\n
- Make decisions based on facts and numbers\n
- Plan and organize long-term\n
- Focus on details and accuracy\n
- Work independently with decision-making freedom`,
    
    Creative: `Creative / Innovative Thinker – loves experimenting and new ideas.\n
Traits of suitable roles:\n
- Generate new and innovative ideas\n
- Experiment with different solutions and adapt quickly\n
- Think outside the box\n
- Enjoy flexible, non-routine environments\n
- Comfortable with calculated risks and trying new approaches`,
    
    Balanced: `Balanced / Organized Thinker – combines planning with adaptability.\n
Traits of suitable roles:\n
- Combine analytical and creative thinking for balanced decisions\n
- Organize projects and follow details precisely\n
- Work well in teams and communicate effectively\n
- Manage time and resources efficiently\n
- Maintain stability while adapting when necessary`
  };

  document.getElementById('resultTitle').innerText = username + " – " + maxType + " Thinker";
  document.getElementById('resultDescription').innerText = descriptions[maxType];
  document.getElementById('quizContainer').style.display = 'none';
  document.getElementById('resultCard').style.display = 'block';
}

// Canvas & Web Share API
async function generateShareImageStory() {
  const canvas = document.getElementById('resultCanvas');
  canvas.width = 1080;
  canvas.height = 1920;
  const ctx = canvas.getContext('2d');

  const gradient = ctx.createLinearGradient(0,0,1080,1920);
  gradient.addColorStop(0, "#080808");
  gradient.addColorStop(1, "#1a1a1a");
  ctx.fillStyle = gradient;
  ctx.fillRect(0,0,1080,1920);

  let logo = new Image();
  logo.src = "YOUR_LOGO_URL_HERE";
  logo.onload = () => {
    ctx.drawImage(logo, 840, 40, 200, 100);
    ctx.fillStyle = "#f7b31e";
    ctx.font = "28px Poppins";
    ctx.fillText("Powered by Golden Developer Technology Systems", 60, 160);

    ctx.fillStyle = "#f49c04";
    ctx.font = "bold 72px Poppins";
    wrapText(ctx, document.getElementById('resultTitle').innerText, 60, 300, 960, 80);

    ctx.fillStyle = "#e5e4e2";
    ctx.font = "36px Poppins";
    wrapText(ctx, document.getElementById('resultDescription').innerText, 60, 500, 960, 50);

    const imageFile = dataURLtoFile(canvas.toDataURL('image/png'), username+'_MindType.png');

    if(navigator.canShare && navigator.canShare({ files: [imageFile] })) {
      navigator.share({ files:[imageFile], title:'My Mind Type', text: username+"'s Personality Result" })
      .catch(err => alert('Sharing failed: '+err));
    } else {
      const link = document.createElement('a');
      link.download = username+"_MindType.png";
      link.href = canvas.toDataURL('image/png');
      link.click();
      alert("Your Story image is ready! Upload manually to Instagram, Facebook, WhatsApp, LinkedIn, etc.");
    }
  };
}

function wrapText(ctx, text, x, y, maxWidth, lineHeight) {
  const words = text.split(' ');
  let line = '';
  for(let n=0; n<words.length; n++){
    let testLine = line + words[n] + ' ';
    let metrics = ctx.measureText(testLine);
    if(metrics.width > maxWidth && n>0){
      ctx.fillText(line, x, y);
      line = words[n] + ' ';
      y += lineHeight;
    } else { line = testLine; }
  }
  ctx.fillText(line, x, y);
}

function dataURLtoFile(dataurl, filename) {
  let arr = dataurl.split(','), mime = arr[0].match(/:(.*?);/)[1],
      bstr = atob(arr[1]), n = bstr.length, u8arr = new Uint8Array(n);
  while(n--){ u8arr[n] = bstr.charCodeAt(n); }
  return new File([u8arr], filename, {type:mime});
}
